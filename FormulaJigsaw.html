        function init() {

            renderBench('cations-bench', cations, 'cation');

            renderBench('anions-bench', anions, 'anion');

            window.addEventListener('mousemove', handleDrag);

            window.addEventListener('mouseup', endDrag);

            window.addEventListener('touchend', endDrag);

        }


        function createPieceSVG(type, charge, symbol) {

            const width = 80;

            const height = charge * 45;

            const holeRadius = 14;

            const color = type === 'cation' ? '#3b82f6' : '#ef4444';

            const sign = type === 'cation' ? '+' : '-';

            const chargeText = charge > 1 ? `${charge}${sign}` : sign;

            

            let svg = `<svg width="${width + 20}" height="${height}" viewBox="0 0 ${width + 20} ${height}" class="svg-piece" xmlns="http://www.w3.org/2000/svg">`;

            

            if (type === 'cation') {

                let d = `M 0 0 H ${width} `;

                for(let i=0; i<charge; i++) {

                    const y = i * 45 + 22.5;

                    d += `V ${y-holeRadius} A ${holeRadius} ${holeRadius} 0 0 0 ${width} ${y+holeRadius} `;

                }

                d += `V ${height} H 0 Z`;

                svg += `<path d="${d}" fill="${color}" stroke="rgba(0,0,0,0.15)" stroke-width="1"/>`;

            } else {

                let d = `M 20 0 H ${width+20} V ${height} H 20 `;

                for(let i=charge-1; i>=0; i--) {

                    const y = i * 45 + 22.5;

                    d += `V ${y+holeRadius} A ${holeRadius} ${holeRadius} 0 0 1 20 ${y-holeRadius} `;

                }

                d += `Z`;

                svg += `<path d="${d}" fill="${color}" stroke="rgba(0,0,0,0.15)" stroke-width="1"/>`;

            }


            const centerX = type === 'cation' ? width/2 : width/2 + 20;

            const centerY = height/2;

            svg += `<text x="${centerX - 6}" y="${centerY + 7}" text-anchor="middle" fill="white" font-weight="900" font-size="20" font-family="serif">${symbol}</text>`;

            svg += `<text x="${centerX + (symbol.length > 2 ? 16 : 12)}" y="${centerY - 4}" text-anchor="start" fill="white" fill-opacity="0.8" font-weight="bold" font-size="12" font-family="sans-serif">${chargeText}</text>`;

            svg += `</svg>`;

            return svg;

        }


        function renderBench(id, list, type) {

            const bench = document.getElementById(id);

            list.forEach(ion => {

                const item = document.createElement('div');

                item.className = 'flex flex-col items-center p-2 border border-slate-100 rounded-xl hover:bg-slate-50 cursor-pointer transition-all active:scale-95 bg-white';

                item.innerHTML = `

                    <div class="mb-1 pointer-events-none">${createPieceSVG(type, ion.charge, ion.symbol)}</div>

                    <span class="text-[9px] font-black text-slate-400 uppercase tracking-tight">${ion.name}</span>

                `;

                item.addEventListener('mousedown', (e) => startSpawn(e, ion, type));

                item.addEventListener('touchstart', (e) => startSpawn(e, ion, type), { passive: false });

                bench.appendChild(item);

            });

        }


        function startSpawn(e, ion, type) {

            if (e.cancelable) e.preventDefault();

            const clientX = e.touches ? e.touches[0].clientX : e.clientX;

            const clientY = e.touches ? e.touches[0].clientY : e.clientY;


            const piece = document.createElement('div');

            piece.className = 'ion-piece';

            piece.dataset.type = type;

            piece.dataset.charge = ion.charge;

            piece.dataset.symbol = ion.symbol;

            piece.dataset.name = ion.name;

            piece.dataset.id = ion.id;

            piece.dataset.poly = ion.poly || false;

            piece.innerHTML = createPieceSVG(type, ion.charge, ion.symbol);

            

            const canvas = document.getElementById('canvas');

            canvas.appendChild(piece);

            const canvasRect = canvas.getBoundingClientRect();

            

            initialX = clientX - canvasRect.left - 50;

            initialY = clientY - canvasRect.top - (ion.charge * 45 / 2);

            

            piece.style.left = `${initialX}px`;

            piece.style.top = `${initialY}px`;

            

            draggedElement = piece;

            startX = clientX;

            startY = clientY;

            activePieces.push(piece);

            document.getElementById('instruction-overlay').style.opacity = '0';


            piece.addEventListener('mousedown', (e) => startExistingDrag(e, piece));

            piece.addEventListener('touchstart', (e) => startExistingDrag(e, piece), { passive: false });

        }


        function startExistingDrag(e, piece) {

            if (e.cancelable) e.preventDefault();

            e.stopPropagation();

            const clientX = e.touches ? e.touches[0].clientX : e.clientX;

            const clientY = e.touches ? e.touches[0].clientY : e.clientY;


            draggedElement = piece;

            startX = clientX;

            startY = clientY;

            initialX = parseFloat(piece.style.left);

            initialY = parseFloat(piece.style.top);

            piece.parentElement.appendChild(piece);

        }


        function handleDrag(e) {

            if (!draggedElement) return;

            if (e.cancelable) e.preventDefault();

            const clientX = e.touches ? e.touches[0].clientX : e.clientX;

            const clientY = e.touches ? e.touches[0].clientY : e.clientY;


            const dx = clientX - startX;

            const dy = clientY - startY;


            draggedElement.style.left = `${initialX + dx}px`;

            draggedElement.style.top = `${initialY + dy}px`;


            snapToOthers(draggedElement);

        }


        function snapToOthers(el) {

            const snapThreshold = 30;

            let currentSnapped = false;


            for (let other of activePieces) {

                if (other === el) continue;

                

                const elType = el.dataset.type;

                const otherType = other.dataset.type;

                const elL = parseFloat(el.style.left);

                const elT = parseFloat(el.style.top);

                const otL = parseFloat(other.style.left);

                const otT = parseFloat(other.style.top);

                const otH = parseInt(other.dataset.charge) * 45;

                const elH = parseInt(el.dataset.charge) * 45;


                // Horizontal snap

                if (elType !== otherType) {

                    const cationX = elType === 'cation' ? elL : otL;

                    const anionX = elType === 'anion' ? elL : otL;

                    const hGap = Math.abs((cationX + 60) - anionX);

                    const vDiff = Math.abs(elT - otT);


                    if (hGap < snapThreshold && vDiff < snapThreshold) {

                        el.style.left = elType === 'cation' ? `${otL - 60}px` : `${otL + 60}px`;

                        el.style.top = `${otT}px`;

                        currentSnapped = true;

                    }

                }


                // Vertical snap

                if (elType === otherType) {

                    const hDiff = Math.abs(elL - otL);

                    const vGapBelow = Math.abs(elT - (otT + otH));

                    const vGapAbove = Math.abs((elT + elH) - otT);


                    if (hDiff < 15) {

                        if (vGapBelow < snapThreshold) {

                            el.style.top = `${otT + otH}px`;

                            el.style.left = `${otL}px`;

                            currentSnapped = true;

                        } else if (vGapAbove < snapThreshold) {

                            el.style.top = `${otT - elH}px`;

                            el.style.left = `${otL}px`;

                            currentSnapped = true;

                        }

                    }

                }

            }

            el.classList.toggle('snapped', currentSnapped);

        }


        function endDrag() {

            draggedElement = null;

        }


        function clearCanvas() {

            activePieces.forEach(p => p.remove());

            activePieces = [];

            document.getElementById('instruction-overlay').style.opacity = '0.2';

        }


        function checkBonding() {

            if (activePieces.length === 0) return;


            const clusters = [];

            const visited = new Set();


            activePieces.forEach((piece, index) => {

                if (visited.has(index)) return;

                

                const cluster = [];

                const queue = [index];

                visited.add(index);


                while(queue.length > 0) {

                    const currIdx = queue.shift();

                    cluster.push(activePieces[currIdx]);


                    activePieces.forEach((other, otherIdx) => {

                        if (visited.has(otherIdx)) return;

                        const r1 = activePieces[currIdx].getBoundingClientRect();

                        const r2 = other.getBoundingClientRect();

                        const touching = !(r2.left > r1.right + 5 || r2.right < r1.left - 5 || r2.top > r1.bottom + 5 || r2.bottom < r1.top - 5);

                        if (touching) {

                            visited.add(otherIdx);

                            queue.push(otherIdx);

                        }

                    });

                }

                clusters.push(cluster);

            });


            let allValid = true;

            let totalCations = 0, totalAnions = 0;

            let cationRef = null, anionRef = null;


            clusters.forEach(cluster => {

                let clusterPos = 0, clusterNeg = 0;

                cluster.forEach(p => {

                    if (p.dataset.type === 'cation') {

                        clusterPos += parseInt(p.dataset.charge);

                        totalCations++;

                        cationRef = p.dataset;

                    } else {

                        clusterNeg += parseInt(p.dataset.charge);

                        totalAnions++;

                        anionRef = p.dataset;

                    }

                });

                if (clusterPos !== clusterNeg || clusterPos === 0) allValid = false;

            });


            if (!allValid || clusters.length === 0) {

                const totalP = activePieces.reduce((acc, p) => acc + (p.dataset.type === 'cation' ? parseInt(p.dataset.charge) : 0), 0);

                const totalN = activePieces.reduce((acc, p) => acc + (p.dataset.type === 'anion' ? parseInt(p.dataset.charge) : 0), 0);

                const diff = totalP - totalN;

                showMessage(diff > 0 ? `Unbalanced: Needs ${diff} more (-) charge.` : (diff < 0 ? `Unbalanced: Needs ${Math.abs(diff)} more (+) charge.` : "Ensure all pieces are connected!"));

                return;

            }


            if (!cationRef || !anionRef) {

                showMessage("Bonding requires both metal/polyatomic cations and anions.");

                return;

            }


            const gcd = (a, b) => b ? gcd(b, a % b) : a;

            const common = gcd(totalCations, totalAnions);

            generateFormula(cationRef, anionRef, totalCations / common, totalAnions / common);

        }


        function formatSymbol(sym) {

            return sym.replace(/([0-9]+)/g, '<sub>$1</sub>');

        }


        function generateFormula(cat, ani, nCat, nAni) {

            let formulaHtml = "";

            

            // Cation portion

            if (cat.poly === "true" && nCat > 1) {

                formulaHtml += `(${formatSymbol(cat.symbol)})<sub>${nCat}</sub>`;

            } else {

                formulaHtml += formatSymbol(cat.symbol) + (nCat > 1 ? `<sub>${nCat}</sub>` : '');

            }


            // Anion portion

            if (ani.poly === "true" && nAni > 1) {

                formulaHtml += `(${formatSymbol(ani.symbol)})<sub>${nAni}</sub>`;

            } else {

                formulaHtml += formatSymbol(ani.symbol) + (nAni > 1 ? `<sub>${nAni}</sub>` : '');

            }


            document.getElementById('compound-name').innerText = `${cat.name} ${ani.name}`;

            document.getElementById('compound-formula').innerHTML = formulaHtml;

            document.getElementById('formula-modal').style.display = 'flex';

        }
